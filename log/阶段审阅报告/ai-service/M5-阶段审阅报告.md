```python
# search_service.py L231-L275
async def hybrid_search(self, query, top_k=None, filters=None):
    candidate_k = top_k * 2
    semantic_results, keyword_results = await asyncio.gather(
        self.search(query, top_k=candidate_k, filters=filters),
        self.keyword_search(query, top_k=candidate_k, filters=filters),
    )
    fused = self._reciprocal_rank_fusion(semantic_results, keyword_results, k=self.rrf_k)
    if self.reranker is not None:
        fused = await self.reranker.rerank(query, fused)
    return fused[:top_k]
```

### 9.2 RRF融合核心代码

```python
# search_service.py L192-L229
def _reciprocal_rank_fusion(self, list1, list2, k=None):
    if k is None:
        k = self.rrf_k
    scores, items = {}, {}
    for rank, item in enumerate(list1):
        pid = item.get("paper_id") or item.get("paperId", "")
        if not pid: continue
        scores[pid] = scores.get(pid, 0.0) + 1.0 / (k + rank + 1)
        if pid not in items: items[pid] = dict(item)
    # ... list2 同理
    sorted_pids = sorted(scores.keys(), key=lambda x: scores[x], reverse=True)
    return [{**items[pid], "rrf_score": scores[pid]} for pid in sorted_pids]
```

```python
def _detect_conflicts(self, text1, text2):
    conflicts = []
    # 1. 数值矛盾：A 增 B 减 或 A 减 B 增
    inc1 = self._NUMERIC_INCREASE_PATTERN.findall(t1)
    dec1 = self._NUMERIC_DECREASE_PATTERN.findall(t1)
    if (inc1 and dec2) or (dec1 and inc2):
        conflicts.append({"type": "numeric", ...})
    # 2. 方向矛盾：A 正向 B 负向 或 A 负向 B 正向
    for pos_words, neg_words in self._DIRECTION_PAIRS:
        if (a_pos and b_neg) or (a_neg and b_pos):
            conflicts.append({"type": "directional", ...})
    # 3. 关键词矛盾（向后兼容）
    kw_conflicts = self._detect_conflict_keywords(t1, t2)
    if kw_conflicts:
        conflicts.append({"type": "keyword", ...})
    return conflicts
```

### 9.4 流式输出核心代码

```python
# llm_service.py L458-L505
async def generate_stream(self, prompt, max_tokens=2048, temperature=0.7):
    stream_start = time.perf_counter()
    first_token_yielded = False
    try:
        async for token in self.active_provider.generate_stream(prompt, max_tokens, temperature):
            if not first_token_yielded:
                first_token_latency_ms = (time.perf_counter() - stream_start) * 1000
                logger.info(f"first_token_latency_ms={first_token_latency_ms:.1f}")
                first_token_yielded = True
            yield token
    except Exception as e:
        # 流式失败降级为非流式
        await self._fallback()
        full_response = await asyncio.wait_for(
            self.active_provider.generate(prompt, max_tokens, temperature), timeout=30)
        yield full_response
```
